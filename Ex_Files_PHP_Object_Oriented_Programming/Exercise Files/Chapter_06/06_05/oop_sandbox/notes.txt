Assignment by Reference

$a = 1;
$b = $a;
$a = 2;
echo $b;
// 1

$a = 1;
$b = &$a;   // $b =& $a; works too
$a = 2;
echo $b;
// 2

- Objects are always assigned by reference 
- Variables reference object identifier
- Assignment is copying the identifier value

$a = new Product;
$a->name = 'coffee';
$b = $a;
$a->name = 'tea';
echo $b->name;
// tea

$a = new Product;
$a->name = 'coffee';
$b = clone $a;
$a->name = 'tea';
echo $b->name;
// coffee
